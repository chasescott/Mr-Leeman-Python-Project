340      def writeString(self, fileName,insertString): 
341          doc = SimpleDocTemplate(fileName,pagesize=letter, 
342                                  rightMargin=72,leftMargin=72, 
343                                  topMargin=72,bottomMargin=18) 
344          Story=[] 
345          logo = "../.idea/images/school.gif" 
346   
347          formatted_time = time.ctime() 
348   
349          im = Image(logo, 2*inch, 2*inch) 
350          Story.append(im) 
351   
352          styles=getSampleStyleSheet() 
353          styles.add(ParagraphStyle(name='Justify', alignment=TA_JUSTIFY)) 
354          ptext = '<font size=12>%s</font>' % formatted_time 
355   
356          Story.append(Paragraph(ptext, styles["Normal"])) 
357          Story.append(Spacer(1, 12)) 
358   
359          ptext = insertString 
360   
361          Story.append(Paragraph(ptext, styles["Justify"])) 
362          Story.append(Spacer(1, 12)) 
363   
364          ptext = '<font size=12>--- END OF REPORT ---</font>' 
365          Story.append(Paragraph(ptext, styles["Normal"])) 
366          Story.append(Spacer(1, 12)) 
367          doc.build(Story) 
368          if os.name == 'nt': 
369              os.system("start "+fileName) 
370          elif os.name == 'posix': 
371              os.system("open "+fileName) 
372          elif os.name == 'mac': 
373              os.system("open "+fileName) 
374          else: 
375              os.sysconf("open "+fileName)
