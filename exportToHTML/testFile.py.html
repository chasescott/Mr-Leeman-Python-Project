116  def writeString(fileName,insertString): 
117          doc = SimpleDocTemplate(fileName,pagesize=letter, 
118                                  rightMargin=72,leftMargin=72, 
119                                  topMargin=72,bottomMargin=18) 
120          Story=[] 
121          logo = "../.idea/images/school.gif" 
122   
123          formatted_time = time.ctime() 
124   
125          im = Image(logo, 2*inch, 2*inch) 
126          Story.append(im) 
127   
128          styles=getSampleStyleSheet() 
129          styles.add(ParagraphStyle(name='Justify', alignment=TA_JUSTIFY)) 
130          ptext = '<font size=12>%s</font>' % formatted_time 
131   
132          Story.append(Paragraph(ptext, styles["Normal"])) 
133          Story.append(Spacer(1, 12)) 
134   
135          ptext = insertString 
136   
137          Story.append(Paragraph(ptext, styles["Justify"])) 
138          Story.append(Spacer(1, 12)) 
139   
140          ptext = '<font size=12>--- END OF REPORT ---</font>' 
141          Story.append(Paragraph(ptext, styles["Normal"])) 
142          Story.append(Spacer(1, 12)) 
143          doc.build(Story) 
144          if os.name == 'nt': 
145                  os.system("start "+fileName) 
146          elif os.name == 'posix': 
147                  os.system("open "+fileName) 
148          elif os.name == 'mac': 
149                  os.system("open "+fileName) 
150          else: 
151                  os.sysconf("open "+fileName)
